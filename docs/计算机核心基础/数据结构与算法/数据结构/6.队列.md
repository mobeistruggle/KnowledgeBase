# 队列

> **queue：**先进先出（FIFO），只允许在队头删除数据，在队尾插入数据

- 队列
  - 队头：最先进来的元素
  - 队尾：最后进来的元素
  - 空队列：队列中没有元素
  - 进队：在队尾中插入数据的过程
  - 出队：在队头中获取后删除数据的过程

- 双端队列
  - 队列的两端都能进行操作

- 循环队列：特殊队列，将队列的队头与头尾进行相连
  - 队空条件：比较头指针和尾指针是否一致的条件
  - 队满条件：比较头指针和队列数量是否一致的条件


## 内置队列

> 仅限特定编程，如python

### 基础队列

> 先进先出队列

```python
from queue import Queue

# 参数是指定队列数量，默认无限大
queue = Queue()

# 入队
queue.put()

# 出队
queue.get()

# 入队时，无需等待
queue.put_nowait()

# 出队时，无需等待
queue.get_nowait()

# 检测队列是否已满
queue.full()

# 检测队列是否为空
queue.empty()
```

### 优先级队列

> 可以给队列设置优先级，根据优先级最顶先后顺序

```python
from queque import PriorityQueue

queue = PriorityQueue()

# 数字越小，优先级越高
queue.put(('优先级数字', '数据',))
```

### 后进先出队列

> 修改模块其他一样，模块为LifoQueue

```python
from queue import LifoQueue

# 其他功能一致
queue = LifoQueue()
```

## 自定义队列

> 为了在特定场景下，实现特定功能

```python
class MyQueue:
    """
    在队列基础上扩展，添加符合特定项目的功能
    """
    def __init__(self, maxsize=0):
        # 初始队列
        self.queue = []
        self.maxsize = maxsize

    def put(self, item):
        if self.is_full():
            return
        self.queue.append(item)

    def get(self):
        if self.is_empty():
            return None
        data = self.queue.pop(-1)
        return data

    def empty(self):
        self.queue.clear()

    def is_empty(self):
        is_empty = len(self.queue) == 0
        return is_empty

    def is_full(self):
        is_full = self.qsize() == self.maxsize
        return is_full

    def qsize(self):
        size = len(self.queue)
        return size
```

## 案例

### 击鼓传花

> 使用队列完成击鼓传花游戏

```javascript
class Queue {

    constructor(size = 1) {
        this.item = []
        this.size = size
    }

    /**
     * 
     * @returns {Number} 队列的大小
     */
    qsize() {
        return this.item.length
    }

    /**
     * 
     * @returns {Boolean} 队列是否排满
     */
    is_full() {
        return this.qsize() === this.size
    }

    is_empty() {
        return this.qsize() === null
    }

    /**
     * 
     * @returns {String} 队列被删除的元素
     */
    get() {
        let res = this.item.shift()
        return res
    }

    /**
     * 
     * @param {*} ele 需要添加到队列的元素
     */
    put(ele) {
        if (this.qsize() === this.size) return console.log('队列已满');
        this.item.push(ele)
    }

    /**
     * 
     * @returns { String } 将数组转成字符串
     */
    toString(){
        return ''.concat(this.item)
    }

}


/**
 * 一个基于队列实现的击鼓传花游戏
 * @param { Array } players 所有参加游戏的玩家 
 * @returns 
 */
function game(players) {
    const queue = new Queue(10);

    // 将所有玩家添加到队列里
    players.forEach(element => {
        queue.put(element)
        if (queue.is_full()) return
    });

    console.log('已加入游戏', queue.item);
    console.log(queue.toString());

    let round = 1
    while (queue.qsize() > 1) {
        console.log(`-----------第${round}轮游戏开始------`);

        // 随机生成敲鼓次数
        gu_count = Math.round(Math.random() * 20)
        for (let i = 1; i < gu_count; i++) {
            // 移出队列并放回队列
            let ele = queue.get()
            queue.put(ele)
        }
        let player = queue.get()
        console.log(`玩家${player}已淘汰！`);
        console.log(`剩余玩家`, queue.item);
        round ++
    }
    let win = queue.get()
    console.log('赢家：', win);
    return win
}

player_arr = ['小明', '小刚', '王强', '赵敏', '硫酸']

game(player_arr)
```

### 回文判断

> 使用循环队列判断输入的文本是否为回文

```js
class DoubleQueue {

    constructor(size = 1) {
        this.item = []
        this.size = size
    }

    /**
     * 
     * @returns {Number} 队列的大小
     */
    qsize() {
        return this.item.length
    }

    /**
     * 
     * @returns {Boolean} 队列是否排满
     */
    is_full() {
        return this.qsize() === this.size
    }

    is_empty() {
        return this.qsize() === null
    }

    /**
     * 从队头出队
     * @returns {String} 队列被删除的元素
     */
    getFront() {
        let res = this.item.shift()
        return res
    }

    /**
     * 从队尾出队
     * @returns {String} 队列被删除的元素
     */
    getLast() {
        let res = this.item.pop()
        return res
    }

    /**
     * 将元素放到队尾
     * @param {*} ele 需要放进队列的元素
     */
    putLast(ele) {
        if (this.qsize() === this.size) return console.log('队列已满');
        this.item.push(ele)
    }

    /**
     * 将元素放到队头
     * @param {*} ele 需要放进队列的元素
     */
    putFront(ele) {
        if (this.qsize() === this.size) return console.log('队列已满');
        this.item.splice(0, 0, ele)
    }

    /**
     * 
     * @returns { String } 将数组转成字符串
     */
    toString(){
        return ''.concat(this.item)
    }

}


/**
 * 校验文本是否为回文 
 * @param { Array } player 玩家
 * @returns { Boolean } 返回是否为回文
*/
function is_palindromic(text) {
    const DQueue = new DoubleQueue(10);
    
    // 去空格
    text = text.replaceAll(' ', '')

    // 将字符放进队列
    let center_item = Math.floor(text.length / 2)
    for (let i=0; i<text.length; i++) {
        if (text[i] === ' ') continue
        DQueue.putLast(text[i])
    }
    console.log(DQueue.item);
    
    for (let i=0; i<center_item; i++) {
        console.log(`正在比对第${i + 1}组`);
        let front = DQueue.getFront()
        let last = DQueue.getLast()
        if (front !== last) return false
    }
    return true
}

let text = 'd a a d'
console.log(is_palindromic(text));
```
