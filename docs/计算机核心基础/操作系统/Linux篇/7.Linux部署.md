### 部署python程序

```shell
# 准备编译程序
yum install gcc patch libffi-devel python-devel zlib-devel bzip2-devel openssl-devel ncurses-devel sqlit-devel readline-devel tk-devel gdbm-devel db4-devel libpcap-devel xz-devel python36-devel -y

# 下载 wget源
yum install wget

# 使用wget源下载python
wget python版本的文件地址

# 解压下载的python源代码压缩包
tar xvf python-3.9.11.tgz

# 安装python程序 程序名为configure
./configure --prefix='指定安装路径'

# 编译源码
make && make install

# 配置环境变量
vi /etc/profile

# 添加如下配置项：
export PYTHON_HOME='python安装后的路径'
export PATH=$PYTHON_HOME/bin:$PATH

# yng'y环境变量
source /etc/profile
```
